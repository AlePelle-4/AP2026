cmake_minimum_required(VERSION 3.20)
project(esercitazione10)

set(CMAKE_CXX_STANDARD 20)
set(CMAKE_CXX_STANDARD_REQUIRED ON)
set(CMAKE_CXX_FLAGS "${CMAKE_CXX_FLAGS} -Wall -Wshadow -Werror=vla")

enable_testing()

find_package(Eigen3 REQUIRED)

add_executable(test_gradiente_coniugato test_gradiente_coniugato.cpp)

target_include_directories(test_gradiente_coniugato PRIVATE 
    ${EIGEN3_INCLUDE_DIRS}
    ${CMAKE_CURRENT_SOURCE_DIR}
)

target_link_libraries(test_gradiente_coniugato PRIVATE Eigen3::Eigen)

add_test(NAME gradiente_coniugato COMMAND test_gradiente_coniugato)
